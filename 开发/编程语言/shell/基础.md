### 条件测试与比较

if

```
# 简单
if [ $# == 1 ]; then
	echo "1"
fi

#或者是
if [ $# == 1 ]
then
	echo "1"
fi
# else
if [ $# == 1 ]; then
	echo "1"
else
	echo "2"
fi

# elif
if [ $# == 1 ]; then
	echo "1"
elif [ $# == 1 ]; then
	echo "2"
else
	echo "3"
fi
```



case

```
case $# in
	4)
		echo "1" 
	;;
	3)
		echo "1" 
	;;
	*)
		echo "*"
	;;
esac
```

while

```
# 样式1: 条件一直为 true，死循环
while true
do
    echo $1
    sleep 2
    
done

# 样式2: 按行读入文件
while read x
do
    echo $x
    sleep 2
    
done<list
```
